        function generateAuditLogs(operator) {
            if (operator.errors === 0) {
                return `<div class="log-status-box" id="log-box" style="color: var(--green); border-color: var(--border-color);">
                            ✨ FLUENT INTEGRITY: NO LOGGED INFRACTIONS RECORDED ✨
                        </div>`;
            }

            let htmlOutput = `<div class="log-container-wrapper">`;
            let remainingErrors = operator.errors;

            if (remainingErrors > 0) {
                // Determine high split counts based on error volume
                let criticalCount = Math.ceil(remainingErrors * 0.6);
                let driftCount = remainingErrors - criticalCount;

                htmlOutput += `
                    <div class="log-row-item">
                        <span class="c-red">CRITICAL POLICY DRIFT // DATA MISALIGNMENT</span>
                        <span class="c-red">${String(criticalCount).padStart(2, '0')}</span>
                    </div>`;
                
                if (driftCount > 0) {
                    htmlOutput += `
                        <div class="log-row-item">
                            <span style="color: var(--orange)">VALIDATION DROPS // UNCLASSIFIED EXCEPTION</span>
                            <span style="color: var(--orange)">${String(driftCount).padStart(2, '0')}</span>
                        </div>`;
                }
            }
            htmlOutput += `</div>`;
            return htmlOutput;
        }

        function renderLeaderboard(filterText = "") {
            listContainer.innerHTML = "";
            let standardMatchFound = false;

            squadData.forEach((item) => {
                if (!item.name.toLowerCase().includes(filterText.toLowerCase())) return;

                const row = document.createElement('div');
                const currentActiveName = document.getElementById('p-name').innerText.replace('// ', '');
                const isItemActive = currentActiveName === item.name || (!currentActiveName && !standardMatchFound);
                
                row.className = `leaderboard-item ${isItemActive ? 'active' : ''}`;
                row.innerHTML = `
                    <span>${item.name}</span>
                    <span class="score">${item.accuracy.toFixed(2)}%</span>
                `;
                
                row.addEventListener('click', () => {
                    document.querySelectorAll('.leaderboard-item').forEach(r => r.classList.remove('active'));
                    row.classList.add('active');
                    updateDashboard(item);
                });

                listContainer.appendChild(row);
                if (isItemActive) standardMatchFound = true;
            });
        }
